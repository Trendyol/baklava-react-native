## Usage
Add it to root view so that it can be seen above all views.
```jsx
// ...
import { theme, TooltipProvider } from '@trendyol/baklava-react-native';
// ...

<ThemeProvider theme={theme}>
  <TooltipProvider>
    <SafeAreaProvider>
      {/* other components or providers. ex: <App/> */}
    </SafeAreaProvider>
  </TooltipProvider>
</ThemeProvider>

// ...
```


```jsx
<Tooltip
  id="basic"
  position="top"
  overlay
  hole
  highlight
  content="Lorem ipsum dolor sit amet">
  <Icon name="info" color="primaryColor" size="m" />
</Tooltip>
```

<br />

## Props

#### id;
Type: `string`
#### content
Type: `string`
#### position
Type: `top`,`bottom`,
#### overlay?
Type: `boolean`
#### hole?
Type: `boolean`
#### holePadding?
Type: `number`
#### highlight?
Type: `boolean`
#### pressable?
Type: `boolean`
#### color?
Type: `string`

The color of the Icon. 
It can be <a href="/docs/design-system-colors--page" title="title">theme color key</a> or <a href="https://reactnative.dev/docs/colors#named-colors" title="title">ColorValue</a>.

Type: `(id: string) => void`
#### wrapperStyle?
Type: `ViewStyle`
#### onClose?
Type: `(id: string) => void`
#### onError?
Type: `(id: string) => void`


## Methods

#### show and hide

```jsx

const tooltipRef = React.useRef<TooltipRef>(null);

React.useEffect(() => {
  setTimeout(()=>{
    tooltipRef.current?.show();
    setTimeout(()=>{
      tooltipRef.current?.hide();
    },5000);
  },5000);
},[])

return (
  <Tooltip
    ref={tooltipRef}
    id="basic"
    position="top"
    overlay
    hole
    highlight
    holePadding={8}
    content="Lorem ipsum dolor sit amet">
    <Icon name="info" color="primaryColor" size="m" />
  </Tooltip>
)
```
